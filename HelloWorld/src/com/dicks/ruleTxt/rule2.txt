rule "Default Filter Stock"
	salience 950
	//agenda-group "first"
    when    	
    	$item : Product()
    	$store : Store()
    	eval(!InventoryDAO.getInstance().containProduct($store, $item))
    then
    	System.out.println("store "+ $store.getStoreId()+ " retracted by rule filter stock");
        retract($store);
end
